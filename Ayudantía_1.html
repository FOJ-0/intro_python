#Crear una lista con 3 elementos
my_list = ['hello', 'world', 'I']
print(my_list)

['hello', 'world', 'I']

#Insertar un elemento en una posición particular
my_list.insert(3, 'am')

my_list[0:3]

print(my_list)
#Eliminar por posición
my_list.pop(2)
print(my_list)

['hello', 'world', 'am']
