def bajo_sobre_media(df_input): 
    df_input.loc[df_input['ocupa'] >= df_input['ocupa'].mean(), 'ocup_cond'] = True
    df_input.loc[df_input['ocupa'] <  df_input['ocupa'].mean(), 'ocup_cond'] = False
    return df_input
